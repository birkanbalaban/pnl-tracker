# Ya da systemd service olarak çalıştırma
sudo nano /etc/systemd/system/pnl_tracker.service


[Unit]
Description=Binance PNL Tracker
After=network.target

[Service]
Type=simple
User=your_username
WorkingDirectory=/path/to/your/project
ExecStart=/usr/bin/python3 /path/to/your/project/main.py
Restart=always
RestartSec=3

[Install]
WantedBy=multi-user.target

# Service'i aktif etme
sudo systemctl enable pnl_tracker
sudo systemctl start pnl_tracker
# Durumunu kontrol etme
sudo systemctl status pnl_tracker
